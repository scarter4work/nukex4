**Goal:** Initialize the NukeX v4 project with build system, vendored dependencies, and all core data structures — fully tested and compiling.

- Double-check all numerical code. First question: "Am I sure?"
- No averages to simplify. Distribution fit is the answer (relevant to later phases, but shapes the data structures now).

```cmake
#   PCL_FOUND        - TRUE if PCL SDK found

set(PCL_LIB_DIR "${PCLDIR}/lib/x64")

set(PCL_LIB_NAMES
    PCL-pxi

    zstd-pxi
    zlib-pxi

    lcms-pxi

        NAMES ${lib}

    set(PCL_INCLUDE_DIRS "${PCL_INCLUDE_DIR}")
    message(STATUS "PCL SDK found: ${PCLDIR}")
```

```cmake
#
# Requires:
#   NUKEX_SIGN_KEYS  - Path to .xssk key file
#   NUKEX_SIGN_PASS  - Signing password
#   PIXINSIGHT_DIR   - Path to PixInsight installation (for signing tool)

set(NUKEX_SIGN_PASS "" CACHE STRING "Signing password")

    COMMAND "${PIXINSIGHT_DIR}/bin/PixInsight.sh"
        "--sign-module-file=$<TARGET_FILE:NukeX>"
        "--xssk-file=${NUKEX_SIGN_KEYS}"
        "--xssk-password=${NUKEX_SIGN_PASS}"
    DEPENDS NukeX
    COMMENT "Signing NukeX module"
    VERBATIM
)

# Package: sign + tarball + SHA1 + XRI update + XRI sign
add_custom_target(package
    COMMAND ${CMAKE_COMMAND} -E echo "=== Signing module ==="
    COMMAND "${PIXINSIGHT_DIR}/bin/PixInsight.sh"
        "--sign-module-file=$<TARGET_FILE:NukeX>"
        "--xssk-file=${NUKEX_SIGN_KEYS}"
        "--xssk-password=${NUKEX_SIGN_PASS}"

    COMMAND ${CMAKE_COMMAND} -E echo "=== Creating package tarball ==="
    COMMAND ${CMAKE_COMMAND} -E make_directory "${NUKEX_REPO_DIR}/bin"
    COMMAND ${CMAKE_COMMAND} -E copy "$<TARGET_FILE:NukeX>" "${NUKEX_REPO_DIR}/bin/"
    COMMAND ${CMAKE_COMMAND} -E copy
        "$<TARGET_FILE_DIR:NukeX>/${NUKEX_MODULE_NAME}.xsgn"
        "${NUKEX_REPO_DIR}/bin/"

    COMMAND ${CMAKE_COMMAND} -E echo "=== Computing SHA1 and creating tarball ==="
    COMMAND ${CMAKE_COMMAND} -E chdir "${NUKEX_REPO_DIR}"
        tar czf "${NUKEX_REPO_DIR}/$<TARGET_FILE_NAME:NukeX>.tar.gz"
        -C "${NUKEX_REPO_DIR}" bin/

    COMMAND ${CMAKE_COMMAND} -E echo "=== Signing updates.xri ==="
    COMMAND "${PIXINSIGHT_DIR}/bin/PixInsight.sh"
        "--sign-xml-file=${NUKEX_REPO_DIR}/updates.xri"
        "--xssk-file=${NUKEX_SIGN_KEYS}"
        "--xssk-password=${NUKEX_SIGN_PASS}"

    DEPENDS NukeX
    COMMENT "Full package: sign module + create tarball + sign XRI"
```

## Task 4: WelfordAccumulator — Test and Implementation

**Files:**
- Create: `src/lib/core/include/nukex/core/welford.hpp`
- Replace: `test/unit/core/test_welford.cpp`

The Welford online algorithm computes running mean and variance in a single pass
with numerical stability. Reference: Welford 1962, "Note on a method for calculating
corrected sums of squares and products." Technometrics 4(3):419-420.

The key insight: naive variance `Σ(x²)/n - (Σx/n)²` suffers catastrophic cancellation
when the mean is large relative to the standard deviation. Welford's recurrence avoids
this by tracking deviations from a running mean.

- [ ] **Step 1: Write WelfordAccumulator tests**

Replace file `test/unit/core/test_welford.cpp`:

```cpp
#include "catch_amalgamated.hpp"
#include "nukex/core/welford.hpp"
#include <cmath>
#include <vector>
#include <numeric>

using nukex::WelfordAccumulator;

TEST_CASE("WelfordAccumulator: default state is zero/empty", "[welford]") {
    WelfordAccumulator w;
    REQUIRE(w.count() == 0);
    REQUIRE(w.mean == 0.0f);
    REQUIRE(w.variance() == 0.0f);
    REQUIRE(w.std_dev() == 0.0f);
    REQUIRE(w.min_val == FLT_MAX);
    REQUIRE(w.max_val == -FLT_MAX);
}

TEST_CASE("WelfordAccumulator: single value", "[welford]") {
    WelfordAccumulator w;
    w.update(42.0f);

    REQUIRE(w.count() == 1);
    REQUIRE(w.mean == Catch::Approx(42.0f));
    // Variance of a single value is 0 (no spread)
    REQUIRE(w.variance() == 0.0f);
    REQUIRE(w.std_dev() == 0.0f);
    REQUIRE(w.min_val == Catch::Approx(42.0f));
    REQUIRE(w.max_val == Catch::Approx(42.0f));
}

TEST_CASE("WelfordAccumulator: two values — known answer", "[welford]") {
    WelfordAccumulator w;
    w.update(2.0f);
    w.update(4.0f);

    REQUIRE(w.count() == 2);
    REQUIRE(w.mean == Catch::Approx(3.0f));
    // Sample variance of {2, 4}: Σ(x-μ)²/(n-1) = ((2-3)²+(4-3)²)/1 = 2.0
    REQUIRE(w.variance() == Catch::Approx(2.0f));
    REQUIRE(w.std_dev() == Catch::Approx(std::sqrt(2.0f)));
    REQUIRE(w.min_val == Catch::Approx(2.0f));
    REQUIRE(w.max_val == Catch::Approx(4.0f));
}

TEST_CASE("WelfordAccumulator: known dataset {1,2,3,4,5}", "[welford]") {
    WelfordAccumulator w;
    for (float v : {1.0f, 2.0f, 3.0f, 4.0f, 5.0f}) {
        w.update(v);
    }

    REQUIRE(w.count() == 5);
    REQUIRE(w.mean == Catch::Approx(3.0f));
    // Sample variance: Σ(x-3)²/4 = (4+1+0+1+4)/4 = 2.5
    REQUIRE(w.variance() == Catch::Approx(2.5f));
    REQUIRE(w.std_dev() == Catch::Approx(std::sqrt(2.5f)));
    REQUIRE(w.min_val == Catch::Approx(1.0f));
    REQUIRE(w.max_val == Catch::Approx(5.0f));
}

TEST_CASE("WelfordAccumulator: identical values → zero variance", "[welford]") {
    WelfordAccumulator w;
    for (int i = 0; i < 100; i++) {
        w.update(7.5f);
    }

    REQUIRE(w.count() == 100);
    REQUIRE(w.mean == Catch::Approx(7.5f));
    REQUIRE(w.variance() == Catch::Approx(0.0f).margin(1e-10f));
    REQUIRE(w.min_val == Catch::Approx(7.5f));
    REQUIRE(w.max_val == Catch::Approx(7.5f));
}

TEST_CASE("WelfordAccumulator: numerical stability — large offset", "[welford]") {
    // This is the classic test case where naive variance fails.
    // Values are 1e9 + small variations. Naive Σx²/n - (Σx/n)² gives garbage
    // due to catastrophic cancellation. Welford must handle this correctly.
    WelfordAccumulator w;
    const float offset = 1.0e9f;
    std::vector<float> values = {
        offset + 1.0f, offset + 2.0f, offset + 3.0f,
        offset + 4.0f, offset + 5.0f
    };
    for (float v : values) {
        w.update(v);
    }

    REQUIRE(w.count() == 5);
    REQUIRE(w.mean == Catch::Approx(offset + 3.0f).epsilon(1e-6f));
    // Sample variance of the offsets {1,2,3,4,5} is 2.5
    // Welford should give this regardless of the large offset
    REQUIRE(w.variance() == Catch::Approx(2.5f).epsilon(1e-4f));
}

TEST_CASE("WelfordAccumulator: negative values", "[welford]") {
    WelfordAccumulator w;
    w.update(-3.0f);
    w.update(-1.0f);
    w.update(1.0f);
    w.update(3.0f);

    REQUIRE(w.count() == 4);
    REQUIRE(w.mean == Catch::Approx(0.0f));
    // Sample variance: (9+1+1+9)/3 = 20/3 ≈ 6.6667
    REQUIRE(w.variance() == Catch::Approx(20.0f / 3.0f).epsilon(1e-5f));
    REQUIRE(w.min_val == Catch::Approx(-3.0f));
    REQUIRE(w.max_val == Catch::Approx(3.0f));
}

TEST_CASE("WelfordAccumulator: large N (1000 samples)", "[welford]") {
    // Uniform values 0, 1, 2, ..., 999
    // Mean = 499.5
    // Sample variance = (999 * 1000 * 1999) / (6 * 999) = 1000*1999/6 = 83291.667
    // Actually: Var = n(n-1)/12 for {0..n-1} with population variance
    // Sample variance of {0,1,...,999}: Σ(x - 499.5)² / 999
    // = (999 * 1000 * (2*999+1)) / (6 * 999) ... let me just compute:
    // Σ(x-μ)² for x=0..999, μ=499.5: = Σx² - n*μ² = 332833500 - 1000*249500.25
    // = 332833500 - 249500250 = 83333250
    // Sample variance = 83333250 / 999 = 83416.6667
    WelfordAccumulator w;
    for (int i = 0; i < 1000; i++) {
        w.update(static_cast<float>(i));
    }

    REQUIRE(w.count() == 1000);
    REQUIRE(w.mean == Catch::Approx(499.5f).epsilon(1e-5f));
    // For {0,1,...,n-1}: sample variance = n*(n-1)/12 ... wait, let me be precise.
    // Population variance of {0,1,...,n-1} = (n²-1)/12
    // Sample variance (dividing by n-1) = n*(n+1)/12 ... no.
    // Let's just compute: Σ_{i=0}^{999} (i - 499.5)² / 999
    // Σ (i - 499.5)² = Σ i² - 2*499.5*Σi + 1000*499.5²
    // Σ i² (0 to 999) = 999*1000*1999/6 = 332833500
    // Σ i (0 to 999) = 999*1000/2 = 499500
```

```cpp
#pragma once

#include <cstdint>
#include <cfloat>
#include <cmath>

namespace nukex {

/// Welford's online algorithm for numerically stable single-pass mean and variance.
///
/// Reference: Welford B.P. (1962), "Note on a method for calculating corrected
/// sums of squares and products." Technometrics 4(3):419-420.
///
/// Unlike naive variance (Σx²/n - (Σx/n)²), this algorithm avoids catastrophic
/// cancellation when the mean is large relative to the standard deviation.
struct WelfordAccumulator {
    float    mean    = 0.0f;
    float    M2      = 0.0f;    // sum of squared deviations from running mean
    float    min_val = FLT_MAX;
    float    max_val = -FLT_MAX;
    uint32_t n       = 0;

    /// Update the accumulator with a new value.
    /// Numerically stable: each update adjusts the mean incrementally,
    /// then computes the deviation product using both pre- and post-update means.
    void update(float x) {
        n++;
        float delta  = x - mean;
        mean        += delta / static_cast<float>(n);
        float delta2 = x - mean;   // note: uses UPDATED mean
        M2          += delta * delta2;
        min_val      = (x < min_val) ? x : min_val;
        max_val      = (x > max_val) ? x : max_val;
    }

    /// Sample variance (Bessel-corrected, divides by n-1).
    /// Returns 0 for n < 2 (variance undefined for fewer than 2 samples).
    float variance() const {
        return (n > 1) ? M2 / static_cast<float>(n - 1) : 0.0f;
    }

    /// Sample standard deviation.
    float std_dev() const {
        return std::sqrt(variance());
    }

    /// Number of values accumulated.
    uint32_t count() const {
        return n;
    }

    /// Reset to initial empty state.
    void reset() {
        mean    = 0.0f;
        M2      = 0.0f;
        min_val = FLT_MAX;
        max_val = -FLT_MAX;
        n       = 0;
    }
};

} // namespace nukex
```

```cpp
#pragma once

#include <cstdint>
#include <algorithm>

namespace nukex {

/// Fixed-bin histogram for per-pixel distribution shape detection.
///
/// 16 bins is sufficient for discriminating Gaussian, bimodal, skewed, and
/// uniform distributions with sample sizes N ≤ 1000 (reservoir K=64).
/// Range is set from observed data via initialize_range(), then values
/// are binned with clamping at edges.
struct PixelHistogram {
    static constexpr int N_BINS = 16;

    uint32_t bins[N_BINS] = {};
    float    range_min    = 0.0f;
    float    range_max    = 1.0f;

    /// Set the histogram range from observed data.
    /// Extends 5% beyond observed min/max on each side to avoid edge crowding.
    void initialize_range(float observed_min, float observed_max) {
        float span = observed_max - observed_min;
        float margin = 0.05f * span;
        range_min = observed_min - margin;
        range_max = observed_max + margin;

        // Guard against degenerate range (all identical values)
        if (range_max - range_min < 1e-10f) {
            range_min -= 0.5f;
            range_max += 0.5f;
        }
    }

    /// Add a value to the histogram. Clamps to [0, N_BINS-1].
    void update(float x) {
        float span = range_max - range_min;
        int bin = static_cast<int>(
            (x - range_min) / span * static_cast<float>(N_BINS));
        bin = std::clamp(bin, 0, N_BINS - 1);
        bins[bin]++;
    }

    /// Total number of values accumulated.
    uint32_t total_count() const {
        uint32_t total = 0;
        for (int i = 0; i < N_BINS; i++) {
            total += bins[i];
        }
        return total;
    }

    /// Index of the bin with the highest count. Ties broken by lowest index.
    int peak_bin() const {
        int best = 0;
        for (int i = 1; i < N_BINS; i++) {
            if (bins[i] > bins[best]) {
                best = i;
            }
        }
        return best;
    }

    /// Center value of a given bin.
    float bin_center(int bin) const {
        float span = range_max - range_min;
        float bin_width = span / static_cast<float>(N_BINS);
        return range_min + (static_cast<float>(bin) + 0.5f) * bin_width;
    }

    /// Reset all bins to zero. Does not reset range.
    void reset() {
        for (int i = 0; i < N_BINS; i++) {
            bins[i] = 0;
        }
    }
};

} // namespace nukex
```

```cpp
#pragma once

#include <cstdint>
#include <algorithm>

namespace nukex {

/// Reservoir sampling (Vitter 1985) with self-describing samples.
///
/// Retains K representative samples from a stream of N items. Each item in the
/// stream has equal probability K/N of being in the final reservoir. No bias
/// toward early or late items.
///
/// Each retained sample carries its own frame context (gain, read noise, PSF
/// weight, etc.) so that pixel selection can proceed without frame-level lookups.
///
/// Reference: Vitter J.S. (1985), "Random Sampling with a Reservoir."
/// ACM Transactions on Mathematical Software 11(1):37-57.
struct ReservoirSample {
    static constexpr int K = 64;

    /// A single retained sample with full frame context.
    struct Sample {
        float    value              = 0.0f;
        float    frame_weight       = 1.0f;
        float    psf_weight         = 1.0f;
        float    read_noise         = 3.0f;    // electrons
        float    gain               = 1.0f;    // e-/ADU
        float    exposure           = 0.0f;    // seconds
        float    lum_ratio          = 1.0f;    // pixel_value / frame_median
        float    sigma_score        = 0.0f;    // deviation from Z-median in σ
        uint16_t frame_index        = 0;       // arrival order
        bool     is_meridian_flipped = false;
    };

    Sample   samples[K] = {};
    int      count       = 0;   // total stream length (not stored count)

    /// Seed the internal RNG. Must be called before update() for reproducible results.
    void seed(uint64_t s) {
        rng_state = s;
        if (rng_state == 0) rng_state = 1;  // avoid zero state
    }

    /// Add a sample to the reservoir.
    /// For the first K items: always stored.
    /// For item n > K: replace a random stored item with probability K/n.
    void update(const Sample& s) {
        if (count < K) {
            samples[count] = s;
        } else {
            // Generate random index in [0, count]
            int j = static_cast<int>(next_random() % static_cast<uint64_t>(count + 1));
            if (j < K) {
                samples[j] = s;
            }
        }
        count++;
    }

    /// Number of samples actually stored (min of count and K).
    int stored_count() const {
        return (count < K) ? count : K;
    }

    /// Reset reservoir to empty state. Does not reset RNG seed.
    void reset() {
        count = 0;
        for (int i = 0; i < K; i++) {
            samples[i] = Sample{};
        }
    }

private:
    /// Simple xorshift64* PRNG. Fast, reasonable quality, no external deps.
    /// Reference: Marsaglia G. (2003), "Xorshift RNGs." Journal of Statistical
    /// Software 8(14):1-6.
    uint64_t rng_state = 1;

    uint64_t next_random() {
        rng_state ^= rng_state >> 12;
        rng_state ^= rng_state << 25;
        rng_state ^= rng_state >> 27;
        return rng_state * 0x2545F4914F6CDD1DULL;
    }
};

} // namespace nukex
```

```cpp
#pragma once

#include <cstdint>

namespace nukex {

/// Distribution shapes identified by fitting the Z-array at each pixel position.
/// Each shape tells a different story about what happened at that pixel across frames.
enum class DistributionShape : uint8_t {
    GAUSSIAN      = 0,   // Clean signal — single peak, symmetric
    BIMODAL       = 1,   // Two populations (e.g., clear frames vs cloud frames)
    SKEWED_LOW    = 2,   // Left tail — partial cloud attenuation in some frames
    SKEWED_HIGH   = 3,   // Right tail — partial trail contamination in some frames
    SPIKE_OUTLIER = 4,   // Clean Gaussian + isolated spike (satellite/cosmic ray)
    UNIFORM       = 5,   // No structure — noise floor or detector defect
    UNKNOWN       = 6    // Fit failed or N too small
};

/// Human-readable name for a distribution shape.
inline const char* distribution_shape_name(DistributionShape shape) {
    switch (shape) {
        case DistributionShape::GAUSSIAN:      return "GAUSSIAN";
        case DistributionShape::BIMODAL:       return "BIMODAL";
        case DistributionShape::SKEWED_LOW:    return "SKEWED_LOW";
        case DistributionShape::SKEWED_HIGH:   return "SKEWED_HIGH";
        case DistributionShape::SPIKE_OUTLIER: return "SPIKE_OUTLIER";
        case DistributionShape::UNIFORM:       return "UNIFORM";
        case DistributionShape::UNKNOWN:       return "UNKNOWN";
    }
    return "UNKNOWN";
}

/// Parameters for a single Gaussian component.
struct GaussianParams {
    float mu        = 0.0f;   // mean
    float sigma     = 0.0f;   // standard deviation
    float amplitude = 0.0f;   // peak height (for mixture models)
};

/// Parameters for a bimodal (two-Gaussian mixture) distribution.
struct BimodalParams {
    GaussianParams comp1;          // first component (typically the signal peak)
    GaussianParams comp2;          // second component (typically contamination)
    float          mixing_ratio = 0.0f;  // weight of comp1 in [0,1]; comp2 = 1 - mixing_ratio
};

/// Parameters for a skew-normal distribution.
/// The skew-normal extends the Gaussian with a shape parameter α that controls asymmetry.
/// α = 0 → Gaussian. α > 0 → right skew. α < 0 → left skew.
struct SkewNormalParams {
    float mu    = 0.0f;   // location
    float sigma = 0.0f;   // scale
    float alpha = 0.0f;   // shape (skewness)
};

/// Fitted distribution for one channel at one pixel position.
///
/// The fitted curve IS the answer. true_signal_estimate comes directly from the
/// fitted parameters, not from averaging raw samples:
///   GAUSSIAN:      μ from Gaussian fit
///   BIMODAL:       μ of dominant signal peak
///   SKEWED_LOW:    mode of fitted skew-normal
///   SKEWED_HIGH:   mode of fitted skew-normal
///   SPIKE_OUTLIER: μ of clean Gaussian component
///   UNIFORM:       KDE mode (triggers non-parametric re-fit)
///   KDE:           mode of kernel density estimate
struct ZDistribution {
    DistributionShape shape = DistributionShape::UNKNOWN;

    /// Fitted parameters. Interpretation depends on shape.
    union {
        GaussianParams   gaussian;      // GAUSSIAN
        BimodalParams    bimodal;       // BIMODAL
        SkewNormalParams skew_normal;   // SKEWED_LOW, SKEWED_HIGH
        GaussianParams   spike_main;   // SPIKE_OUTLIER: the clean Gaussian component
    } params = {};

    /// Spike information (valid only when shape == SPIKE_OUTLIER)
    float   spike_value       = 0.0f;
    uint8_t spike_frame_index = 0;

    /// Goodness of fit metrics
    float r_squared = 0.0f;    // coefficient of determination
    float aic       = 0.0f;    // Akaike Information Criterion: 2k - 2·ln(L)
    float bic       = 0.0f;    // Bayesian Information Criterion

    /// Signal extraction — derived from the fitted curve parameters
    float true_signal_estimate = 0.0f;   // the pixel value
    float signal_uncertainty   = 0.0f;   // 1σ uncertainty on the estimate
    float confidence           = 0.0f;   // [0,1] trustworthiness of this estimate

    /// Non-parametric fallback (used when all standard models fail AIC threshold)
    float kde_mode      = 0.0f;
    float kde_bandwidth = 0.0f;
    bool  used_nonparametric = false;
};

} // namespace nukex
```

```cpp
#pragma once

#include "nukex/core/types.hpp"
#include "nukex/core/welford.hpp"
#include "nukex/core/histogram.hpp"
#include "nukex/core/reservoir.hpp"
#include "nukex/core/distribution.hpp"
#include <cstdint>

namespace nukex {

/// Voxel flag bits.
namespace VoxelFlags {
    constexpr uint8_t BORDER    = 0x01;  // pixel is at alignment border (partial frame coverage)
    constexpr uint8_t SATURATED = 0x02;  // at least one frame was saturated at this position
    constexpr uint8_t LOW_N     = 0x04;  // fewer frames than expected contributed
    constexpr uint8_t FIT_FAILED = 0x08; // distribution fitting failed on at least one channel
}

/// The data structure of record. One instance per (x,y) pixel position.
///
/// EVERY per-pixel statistic lives in this struct. No parallel arrays.
/// Processing may use temporary shadow structures (SoA views, GPU buffers)
/// for performance, but all computed results are written back here.
/// The voxel is canonical truth.
struct SubcubeVoxel {
    // ── Per-channel accumulated statistics (streaming pass) ──────────
    WelfordAccumulator  welford[MAX_CHANNELS];
    PixelHistogram      histogram[MAX_CHANNELS];
    ReservoirSample     reservoir[MAX_CHANNELS];
    ZDistribution       distribution[MAX_CHANNELS];

    // ── Per-channel output values (from pixel selector) ──────────────
    float               output_value[MAX_CHANNELS]           = {};
    float               noise_sigma[MAX_CHANNELS]            = {};
    float               snr[MAX_CHANNELS]                    = {};

    // ── Per-channel robust statistics ────────────────────────────────
    float               mad[MAX_CHANNELS]                    = {};
    float               biweight_midvariance[MAX_CHANNELS]   = {};
    float               iqr[MAX_CHANNELS]                    = {};

    // ── Classification summary (accumulated across ALL frames) ───────
    uint16_t            cloud_frame_count  = 0;
    uint16_t            trail_frame_count  = 0;
    float               worst_sigma_score  = 0.0f;
    float               best_sigma_score   = 0.0f;
    float               mean_weight        = 0.0f;
    float               total_exposure     = 0.0f;   // sum of contributing exposure seconds

    // ── Cross-channel quality ────────────────────────────────────────
    float               confidence         = 0.0f;   // [0,1] overall pixel trustworthiness
    float               quality_score      = 0.0f;   // composite quality metric
    DistributionShape   dominant_shape     = DistributionShape::UNKNOWN;

    // ── Spatial context ──────────────────────────────────────────────
    float               gradient_mag       = 0.0f;
    float               local_background   = 0.0f;
    float               local_rms          = 0.0f;

    // ── PSF quality at this position ─────────────────────────────────
    float               mean_fwhm          = 0.0f;
    float               mean_eccentricity  = 0.0f;
    float               best_fwhm          = 0.0f;

    // ── Bookkeeping ──────────────────────────────────────────────────
    uint16_t            n_frames           = 0;
    uint8_t             n_channels         = 0;
    uint8_t             flags              = 0;

    // ── Flag operations ──────────────────────────────────────────────

    bool has_flag(uint8_t flag) const { return (flags & flag) != 0; }
    void set_flag(uint8_t flag)       { flags |= flag; }
    void clear_flag(uint8_t flag)     { flags &= ~flag; }
};

} // namespace nukex
```

```cpp
/// The subcube — a 2D grid of voxels, one per (x,y) pixel position.
///
/// Owns the voxel storage and provides spatial access. Frame-level metadata
/// vectors exist transiently during the streaming pass; after streaming,
/// all relevant data has been folded into the voxels.
class Cube {
public:
    int           width  = 0;
    int           height = 0;
    ChannelConfig channel_config;
    int           n_frames_loaded = 0;

    /// Construct a cube with given dimensions and channel configuration.
    /// Allocates all voxels and initializes them with the correct channel count.
    /// Each voxel's reservoirs are seeded with a unique value derived from position.
    Cube(int w, int h, const ChannelConfig& config);

    /// Default constructor for deferred initialization.
    Cube() = default;

    /// Access voxel at (x, y). No bounds checking in release builds.
    SubcubeVoxel&       at(int x, int y)       { return voxels_[y * width + x]; }
    const SubcubeVoxel& at(int x, int y) const { return voxels_[y * width + x]; }

    /// Total number of pixels (width × height).
    int total_pixels() const { return width * height; }

    /// Bounds check.
    bool is_valid_coord(int x, int y) const {
        return x >= 0 && x < width && y >= 0 && y < height;
    }

private:
    std::vector<SubcubeVoxel> voxels_;
};

} // namespace nukex
```

```cpp
#include "nukex/core/cube.hpp"

namespace nukex {

Cube::Cube(int w, int h, const ChannelConfig& config)
    : width(w)
    , height(h)
    , channel_config(config)
    , n_frames_loaded(0)
    , voxels_(static_cast<size_t>(w) * static_cast<size_t>(h))
{
    // Initialize each voxel with the correct channel count and unique reservoir seeds.
    // Seed derivation: combine pixel position into a unique 64-bit value.
    // Using a hash-like combination ensures no two voxels share the same seed,
    // which prevents correlated reservoir replacement patterns.
    for (int y = 0; y < height; y++) {
        for (int x = 0; x < width; x++) {
            SubcubeVoxel& v = voxels_[y * width + x];
            v.n_channels = config.n_channels;

            // Seed each channel's reservoir with a unique value derived from (x, y, ch).
            // FNV-1a inspired mixing for good dispersion.
            for (int ch = 0; ch < config.n_channels; ch++) {
                uint64_t seed = 14695981039346656037ULL;  // FNV offset basis
                seed ^= static_cast<uint64_t>(x);
                seed *= 1099511628211ULL;  // FNV prime
                seed ^= static_cast<uint64_t>(y);
                seed *= 1099511628211ULL;
                seed ^= static_cast<uint64_t>(ch);
                seed *= 1099511628211ULL;
                v.reservoir[ch].seed(seed);
            }
        }
    }
}

} // namespace nukex
```
